189 //----------------------------------------------------------------------------------------------------------------------------------------------------------//
190 // steam callback and call-result helpers
191 //
192 // The following macros and classes are used to register your application for
193 // callbacks and call-results, which are delivered in a predictable manner.
194 //
195 // STEAM_CALLBACK macros are meant for use inside of a C++ class definition.
196 // They map a Steam notification callback directly to a class member function
197 // which is automatically prototyped as "void func( callback_type *pParam )".
198 //
199 // CCallResult is used with specific Steam APIs that return "result handles".
200 // The handle can be passed to a CCallResult object's Set function, along with
201 // an object pointer and member-function pointer. The member function will
202 // be executed once the results of the Steam API call are available.
203 //
204 // CCallback and CCallbackManual classes can be used instead of STEAM_CALLBACK
205 // macros if you require finer control over registration and unregistration.
206 //
207 // Callbacks and call-results are queued automatically and are only
208 // delivered/executed when your application calls SteamAPI_RunCallbacks().
209 //----------------------------------------------------------------------------------------------------------------------------------------------------------//
210 
211 // SteamAPI_RunCallbacks is safe to call from multiple threads simultaneously,
212 // but if you choose to do this, callback code could be executed on any thread.
213 // One alternative is to call SteamAPI_RunCallbacks from the main thread only,
214 // and call SteamAPI_ReleaseCurrentThreadMemory regularly on other threads.
215 S_API void S_CALLTYPE SteamAPI_RunCallbacks();
216 
217 
218 // Declares a callback member function plus a helper member variable which
219 // registers the callback on object creation and unregisters on destruction.
220 // The optional fourth 'var' param exists only for backwards-compatibility
221 // and can be ignored.
222 #define STEAM_CALLBACK( thisclass, func, .../*callback_type, [deprecated] var*/ ) \
223  _STEAM_CALLBACK_SELECT( ( __VA_ARGS__, 4, 3 ), ( , thisclass, func, __VA_ARGS__ ) )
224 
225 // Declares a callback function and a named CCallbackManual variable which
226 // has Register and Unregister functions instead of automatic registration.
227 #define STEAM_CALLBACK_MANUAL( thisclass, func, callback_type, var ) \
228  CCallbackManual< thisclass, callback_type > var; void func( callback_type *pParam )
