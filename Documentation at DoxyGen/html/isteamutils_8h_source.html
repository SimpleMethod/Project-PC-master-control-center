51 //-----------------------------------------------------------------------------
52 // Purpose: interface to user independent utility functions
53 //-----------------------------------------------------------------------------
54 class ISteamUtils
55 {
56 public:
57  // return the number of seconds since the user
58  virtual uint32 GetSecondsSinceAppActive() = 0;
59  virtual uint32 GetSecondsSinceComputerActive() = 0;
60 
61  // the universe this client is connecting to
62  virtual EUniverse GetConnectedUniverse() = 0;
63 
64  // Steam server time. Number of seconds since January 1, 1970, GMT (i.e unix time)
65  virtual uint32 GetServerRealTime() = 0;
66 
67  // returns the 2 digit ISO 3166-1-alpha-2 format country code this client is running in (as looked up via an IP-to-location database)
68  // e.g "US" or "UK".
69  virtual const char *GetIPCountry() = 0;
70 
71  // returns true if the image exists, and valid sizes were filled out
72  virtual bool GetImageSize( int iImage, uint32 *pnWidth, uint32 *pnHeight ) = 0;
73 
74  // returns true if the image exists, and the buffer was successfully filled out
75  // results are returned in RGBA format
76  // the destination buffer size should be 4 * height * width * sizeof(char)
77  virtual bool GetImageRGBA( int iImage, uint8 *pubDest, int nDestBufferSize ) = 0;
78 
79  // returns the IP of the reporting server for valve - currently only used in Source engine games
80  virtual bool GetCSERIPPort( uint32 *unIP, uint16 *usPort ) = 0;
81 
82  // return the amount of battery power left in the current system in % [0..100], 255 for being on AC power
83  virtual uint8 GetCurrentBatteryPower() = 0;
84 
85  // returns the appID of the current process
86  virtual uint32 GetAppID() = 0;
87 
88  // Sets the position where the overlay instance for the currently calling game should show notifications.
89  // This position is per-game and if this function is called from outside of a game context it will do nothing.
90  virtual void SetOverlayNotificationPosition( ENotificationPosition eNotificationPosition ) = 0;
91 
92  // API asynchronous call results
93  // can be used directly, but more commonly used via the callback dispatch API (see steam_api.h)
94  virtual bool IsAPICallCompleted( SteamAPICall_t hSteamAPICall, bool *pbFailed ) = 0;
95  virtual ESteamAPICallFailure GetAPICallFailureReason( SteamAPICall_t hSteamAPICall ) = 0;
96  virtual bool GetAPICallResult( SteamAPICall_t hSteamAPICall, void *pCallback, int cubCallback, int iCallbackExpected, bool *pbFailed ) = 0;
97 
98  // Deprecated. Applications should use SteamAPI_RunCallbacks() instead. Game servers do not need to call this function.
99  STEAM_PRIVATE_API( virtual void RunFrame() = 0; )
100 
101  // returns the number of IPC calls made since the last time this function was called
102  // Used for perf debugging so you can understand how many IPC calls your game makes per frame
103  // Every IPC call is at minimum a thread context switch if not a process one so you want to rate
104  // control how often you do them.
105  virtual uint32 GetIPCCallCount() = 0;
106 
107  // API warning handling
108  // 'int' is the severity; 0 for msg, 1 for warning
109  // 'const char *' is the text of the message
110  // callbacks will occur directly after the API function is called that generated the warning or message
111  virtual void SetWarningMessageHook( SteamAPIWarningMessageHook_t pFunction ) = 0;
112 
113  // Returns true if the overlay is running & the user can access it. The overlay process could take a few seconds to
114  // start & hook the game process, so this function will initially return false while the overlay is loading.
115  virtual bool IsOverlayEnabled() = 0;
116 
117  // Normally this call is unneeded if your game has a constantly running frame loop that calls the
118  // D3D Present API, or OGL SwapBuffers API every frame.
119  //
120  // However, if you have a game that only refreshes the screen on an event driven basis then that can break
121  // the overlay, as it uses your Present/SwapBuffers calls to drive it's internal frame loop and it may also
122  // need to Present() to the screen any time an even needing a notification happens or when the overlay is
123  // brought up over the game by a user. You can use this API to ask the overlay if it currently need a present
124  // in that case, and then you can check for this periodically (roughly 33hz is desirable) and make sure you
125  // refresh the screen with Present or SwapBuffers to allow the overlay to do it's work.
126  virtual bool BOverlayNeedsPresent() = 0;
127 
128  // Asynchronous call to check if an executable file has been signed using the public key set on the signing tab
129  // of the partner site, for example to refuse to load modified executable files.
130  // The result is returned in CheckFileSignature_t.
131  // k_ECheckFileSignatureNoSignaturesFoundForThisApp - This app has not been configured on the signing tab of the partner site to enable this function.
132  // k_ECheckFileSignatureNoSignaturesFoundForThisFile - This file is not listed on the signing tab for the partner site.
133  // k_ECheckFileSignatureFileNotFound - The file does not exist on disk.
134  // k_ECheckFileSignatureInvalidSignature - The file exists, and the signing tab has been set for this file, but the file is either not signed or the signature does not match.
135  // k_ECheckFileSignatureValidSignature - The file is signed and the signature is valid.
136  CALL_RESULT( CheckFileSignature_t )
137  virtual SteamAPICall_t CheckFileSignature( const char *szFileName ) = 0;
138 
139  // Activates the Big Picture text input dialog which only supports gamepad input
140  virtual bool ShowGamepadTextInput( EGamepadTextInputMode eInputMode, EGamepadTextInputLineMode eLineInputMode, const char *pchDescription, uint32 unCharMax, const char *pchExistingText ) = 0;
141 
142  // Returns previously entered text & length
143  virtual uint32 GetEnteredGamepadTextLength() = 0;
144  virtual bool GetEnteredGamepadTextInput( char *pchText, uint32 cchText ) = 0;
145 
146  // returns the language the steam client is running in, you probably want ISteamApps::GetCurrentGameLanguage instead, this is for very special usage cases
147  virtual const char *GetSteamUILanguage() = 0;
148 
149  // returns true if Steam itself is running in VR mode
150  virtual bool IsSteamRunningInVR() = 0;
151 
152  // Sets the inset of the overlay notification from the corner specified by SetOverlayNotificationPosition.
153  virtual void SetOverlayNotificationInset( int nHorizontalInset, int nVerticalInset ) = 0;
154 
155  // returns true if Steam & the Steam Overlay are running in Big Picture mode
156  // Games much be launched through the Steam client to enable the Big Picture overlay. During development,
157  // a game can be added as a non-steam game to the developers library to test this feature
158  virtual bool IsSteamInBigPictureMode() = 0;
159 
160  // ask SteamUI to create and render its OpenVR dashboard
161  virtual void StartVRDashboard() = 0;
162 
163  // Returns true if the HMD content will be streamed via Steam In-Home Streaming
164  virtual bool IsVRHeadsetStreamingEnabled() = 0;
165 
166  // Set whether the HMD content will be streamed via Steam In-Home Streaming
167  // If this is set to true, then the scene in the HMD headset will be streamed, and remote input will not be allowed.
168  // If this is set to false, then the application window will be streamed instead, and remote input will be allowed.
169  // The default is true unless "VRHeadsetStreaming" "0" is in the extended appinfo for a game.
170  // (this is useful for games that have asymmetric multiplayer gameplay)
171  virtual void SetVRHeadsetStreamingEnabled( bool bEnabled ) = 0;
172 };
