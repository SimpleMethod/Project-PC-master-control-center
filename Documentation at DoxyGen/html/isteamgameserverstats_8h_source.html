15 //-----------------------------------------------------------------------------
16 // Purpose: Functions for authenticating users via Steam to play on a game server
17 //-----------------------------------------------------------------------------
18 class ISteamGameServerStats
19 {
20 public:
21  // downloads stats for the user
22  // returns a GSStatsReceived_t callback when completed
23  // if the user has no stats, GSStatsReceived_t.m_eResult will be set to k_EResultFail
24  // these stats will only be auto-updated for clients playing on the server. For other
25  // users you'll need to call RequestUserStats() again to refresh any data
26  CALL_RESULT( GSStatsReceived_t )
27  virtual SteamAPICall_t RequestUserStats( CSteamID steamIDUser ) = 0;
28 
29  // requests stat information for a user, usable after a successful call to RequestUserStats()
30  virtual bool GetUserStat( CSteamID steamIDUser, const char *pchName, int32 *pData ) = 0;
31  virtual bool GetUserStat( CSteamID steamIDUser, const char *pchName, float *pData ) = 0;
32  virtual bool GetUserAchievement( CSteamID steamIDUser, const char *pchName, bool *pbAchieved ) = 0;
33 
34  // Set / update stats and achievements.
35  // Note: These updates will work only on stats game servers are allowed to edit and only for
36  // game servers that have been declared as officially controlled by the game creators.
37  // Set the IP range of your official servers on the Steamworks page
38  virtual bool SetUserStat( CSteamID steamIDUser, const char *pchName, int32 nData ) = 0;
39  virtual bool SetUserStat( CSteamID steamIDUser, const char *pchName, float fData ) = 0;
40  virtual bool UpdateUserAvgRateStat( CSteamID steamIDUser, const char *pchName, float flCountThisSession, double dSessionLength ) = 0;
41 
42  virtual bool SetUserAchievement( CSteamID steamIDUser, const char *pchName ) = 0;
43  virtual bool ClearUserAchievement( CSteamID steamIDUser, const char *pchName ) = 0;
44 
45  // Store the current data on the server, will get a GSStatsStored_t callback when set.
46  //
47  // If the callback has a result of k_EResultInvalidParam, one or more stats
48  // uploaded has been rejected, either because they broke constraints
49  // or were out of date. In this case the server sends back updated values.
50  // The stats should be re-iterated to keep in sync.
51  CALL_RESULT( GSStatsStored_t )
52  virtual SteamAPICall_t StoreUserStats( CSteamID steamIDUser ) = 0;
53 };
