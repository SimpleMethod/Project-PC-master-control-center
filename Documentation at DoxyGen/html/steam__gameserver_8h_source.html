25 // Initialize ISteamGameServer interface object, and set server properties which may not be changed.
26 //
27 // After calling this function, you should set any additional server parameters, and then
28 // call ISteamGameServer::LogOnAnonymous() or ISteamGameServer::LogOn()
29 //
30 // - usSteamPort is the local port used to communicate with the steam servers.
31 // - usGamePort is the port that clients will connect to for gameplay.
32 // - usQueryPort is the port that will manage server browser related duties and info
33 // pings from clients. If you pass MASTERSERVERUPDATERPORT_USEGAMESOCKETSHARE for usQueryPort, then it
34 // will use "GameSocketShare" mode, which means that the game is responsible for sending and receiving
35 // UDP packets for the master server updater. See references to GameSocketShare in isteamgameserver.h.
36 // - The version string is usually in the form x.x.x.x, and is used by the master server to detect when the
37 // server is out of date. (Only servers with the latest version will be listed.)

181 // This function must be declared inline in the header so the module using steam_api.dll gets the version names they want.
182 inline bool CSteamGameServerAPIContext::Init()
183 {
184  HSteamUser hSteamUser = SteamGameServer_GetHSteamUser();
185  HSteamPipe hSteamPipe = SteamGameServer_GetHSteamPipe();
186  if ( !hSteamPipe )
187  return false;
188 
189  m_pSteamClient = (ISteamClient*) SteamInternal_CreateInterface( STEAMCLIENT_INTERFACE_VERSION );
190  if ( !m_pSteamClient )
191  return false;
192 
193  m_pSteamGameServer = m_pSteamClient->GetISteamGameServer( hSteamUser, hSteamPipe, STEAMGAMESERVER_INTERFACE_VERSION );
194  if ( !m_pSteamGameServer )
195  return false;
196 
197  m_pSteamGameServerUtils = m_pSteamClient->GetISteamUtils( hSteamPipe, STEAMUTILS_INTERFACE_VERSION );
198  if ( !m_pSteamGameServerUtils )
199  return false;
200 
201  m_pSteamGameServerNetworking = m_pSteamClient->GetISteamNetworking( hSteamUser, hSteamPipe, STEAMNETWORKING_INTERFACE_VERSION );
202  if ( !m_pSteamGameServerNetworking )
203  return false;
204 
205  m_pSteamGameServerStats = m_pSteamClient->GetISteamGameServerStats( hSteamUser, hSteamPipe, STEAMGAMESERVERSTATS_INTERFACE_VERSION );
206  if ( !m_pSteamGameServerStats )
207  return false;
208 
209  m_pSteamHTTP = m_pSteamClient->GetISteamHTTP( hSteamUser, hSteamPipe, STEAMHTTP_INTERFACE_VERSION );
210  if ( !m_pSteamHTTP )
211  return false;
212 
213  m_pSteamInventory = m_pSteamClient->GetISteamInventory( hSteamUser, hSteamPipe, STEAMINVENTORY_INTERFACE_VERSION );
214  if ( !m_pSteamInventory )
215  return false;
216 
217  m_pSteamUGC = m_pSteamClient->GetISteamUGC( hSteamUser, hSteamPipe, STEAMUGC_INTERFACE_VERSION );
218  if ( !m_pSteamUGC )
219  return false;
220 
221  m_pSteamApps = m_pSteamClient->GetISteamApps( hSteamUser, hSteamPipe, STEAMAPPS_INTERFACE_VERSION );
222  if ( !m_pSteamApps )
223  return false;
224 
225  return true;
226 }
