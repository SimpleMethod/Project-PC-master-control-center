915 inline bool CSteamID::IsValid() const
916 {
917  if ( m_steamid.m_comp.m_EAccountType <= k_EAccountTypeInvalid || m_steamid.m_comp.m_EAccountType >= k_EAccountTypeMax )
918  return false;
919 
920  if ( m_steamid.m_comp.m_EUniverse <= k_EUniverseInvalid || m_steamid.m_comp.m_EUniverse >= k_EUniverseMax )
921  return false;
922 
923  if ( m_steamid.m_comp.m_EAccountType == k_EAccountTypeIndividual )
924  {
925  if ( m_steamid.m_comp.m_unAccountID == 0 || m_steamid.m_comp.m_unAccountInstance > k_unSteamUserWebInstance )
926  return false;
927  }
928 
929  if ( m_steamid.m_comp.m_EAccountType == k_EAccountTypeClan )
930  {
931  if ( m_steamid.m_comp.m_unAccountID == 0 || m_steamid.m_comp.m_unAccountInstance != 0 )
932  return false;
933  }
934 
935  if ( m_steamid.m_comp.m_EAccountType == k_EAccountTypeGameServer )
936  {
937  if ( m_steamid.m_comp.m_unAccountID == 0 )
938  return false;
939  // Any limit on instances? We use them for local users and bots
940  }
941  return true;
942 }
