22 inline CSteamAPIContext& SteamInternal_ModuleContext()
23 {
24  // SteamInternal_ContextInit takes a base pointer for the equivalent of
25  // struct { void (*pFn)(void* pCtx); uintp counter; CSteamAPIContext ctx; }
26  // Do not change layout of 2 + sizeof... or add non-pointer aligned data!
27  // NOTE: declaring "static CSteamAPIConext" creates a large function
28  // which queries the initialization status of the object. We know that
29  // it is pointer-aligned and fully memset with zeros, so just alias a
30  // static buffer of the appropriate size and call it a CSteamAPIContext.
31  static void* s_CallbackCounterAndContext[ 2 + sizeof(CSteamAPIContext)/sizeof(void*) ] = { (void*)&SteamInternal_OnContextInit, 0 };
32  return *(CSteamAPIContext*)SteamInternal_ContextInit( s_CallbackCounterAndContext );
33 }

293 template< class T, class P, bool bGameserver >
294 inline void CCallback< T, P, bGameserver >::Register( T *pObj, func_t func )
295 {
296  if ( !pObj || !func )
297  return;
298 
299  if ( this->m_nCallbackFlags & CCallbackBase::k_ECallbackFlagsRegistered )
300  Unregister();
301 
302  m_pObj = pObj;
303  m_Func = func;
304  // SteamAPI_RegisterCallback sets k_ECallbackFlagsRegistered
305  SteamAPI_RegisterCallback( this, P::k_iCallback );
306 }
