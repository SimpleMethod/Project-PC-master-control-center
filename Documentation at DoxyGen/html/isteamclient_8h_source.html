112 //-----------------------------------------------------------------------------
113 // Purpose: Interface to creating a new steam instance, or to
114 // connect to an existing steam instance, whether it's in a
115 // different process or is local.
116 //
117 // For most scenarios this is all handled automatically via SteamAPI_Init().
118 // You'll only need these APIs if you have a more complex versioning scheme,
119 // or if you want to implement a multiplexed gameserver where a single process
120 // is handling multiple games at once with independent gameserver SteamIDs.
121 //-----------------------------------------------------------------------------
122 class ISteamClient
123 {
124 public:
125  // Creates a communication pipe to the Steam client.
126  // NOT THREADSAFE - ensure that no other threads are accessing Steamworks API when calling
127  virtual HSteamPipe CreateSteamPipe() = 0;
128 
129  // Releases a previously created communications pipe
130  // NOT THREADSAFE - ensure that no other threads are accessing Steamworks API when calling
131  virtual bool BReleaseSteamPipe( HSteamPipe hSteamPipe ) = 0;
132 
133  // connects to an existing global user, failing if none exists
134  // used by the game to coordinate with the steamUI
135  // NOT THREADSAFE - ensure that no other threads are accessing Steamworks API when calling
136  virtual HSteamUser ConnectToGlobalUser( HSteamPipe hSteamPipe ) = 0;
137 
138  // used by game servers, create a steam user that won't be shared with anyone else
139  // NOT THREADSAFE - ensure that no other threads are accessing Steamworks API when calling
140  virtual HSteamUser CreateLocalUser( HSteamPipe *phSteamPipe, EAccountType eAccountType ) = 0;
141 
142  // removes an allocated user
143  // NOT THREADSAFE - ensure that no other threads are accessing Steamworks API when calling
144  virtual void ReleaseUser( HSteamPipe hSteamPipe, HSteamUser hUser ) = 0;
